地形-地形纹理合并问题以及解决方案

Posted on 2020-03-24 | In 地形

地形纹理合并接缝问题:

解决方案:

(一)
开销高,效果完美:

Code:
float blendRatio = tex2D(_BlendTex, i.uv).z;

float2 worldScale = (worldPos.xz * _BlockScale);
float2 worldUv = 0.234375 * frac(worldScale) + 0.0078125;
float2 dx = clamp(0.234375 * ddx(worldScale), -0.0078125, 0.0078125);
float2 dy = clamp(0.234375 * ddy(worldScale), -0.0078125, 0.0078125);

float2 uv0 = worldUv.xy + decodedIndices.xy;
float2 uv1 = worldUv.xy + decodedIndices.zw;
// Sample the two texture
float4 col0 = tex2D(_BlockMainTex, uv0, dx, dy);
float4 col1 = tex2D(_BlockMainTex, uv1, dx, dy);
// Blend the two textures
float4 col = lerp(col0, col1, blendRatio);

参考资料:
https://candycat1992.github.io/2016/11/28/blend-terrain-textures/

(二)
开销低,效果存在瑕疵

Code:
loat blendRatio = tex2D(_BlendTex, i.uv).z;

float2 worldScale = (worldPos.xz * _BlockScale);
float2 worldUv = 0.234375 * frac(worldScale) + 0.0078125;
float depth = i.depth * i.depth * 1/400;
depth = min (depth, 2.5);

float2 uv0 = worldUv.xy + decodedIndices.xy;
float2 uv1 = worldUv.xy + decodedIndices.zw;
// Sample the two texture
half4 colR = tex2Dlod(_BlockMainTex, half4(uv0.xy,0,depth));
half4 colG = tex2Dlod(_BlockMainTex, half4(uv1.zw,0,depth));
// Blend the two textures
float4 col = lerp(col0, col1, blendRatio);

地形纹理合并马赛克问题:

解决方案:

修改ID图生成方式:
Color[] colsRes3L = new Color[width * width];

for (int i = 0; i < width; i++)
{
for (int j = 0; j < width; j++)
{
int index = (j) + (i) * width;
int count = pixelDatas[index].Count;
//8 layer1
for (int k = 0; k < 8; k++)
{
int compareIndex = Mathf.Clamp(index + Mathf.FloorToInt(offsetTile[k].x + offsetTile[k].y * width), 0, 512 * 512 - 1);
bool all_diff = true;
for (int pixelLayerIndex = 0; pixelLayerIndex < count; pixelLayerIndex++)
{
if (pixelDatasReal[compareIndex][0].index == pixelDatas[index][pixelLayerIndex].index)
{
all_diff = false;
break;
}
}
if (all_diff)
{
SplatPixelData temp = pixelDatasReal[compareIndex][0];
pixelDatas[index].Add(temp);
count++;
}
}
//8 layer2
for (int k = 0; k < 8; k++)
{
int compareIndex = Mathf.Clamp(index + Mathf.FloorToInt(offsetTile[k].x + offsetTile[k].y * width), 0, 512 * 512 - 1);
if (pixelDatasReal[compareIndex].Count < 2)
continue;
bool all_diff = true;
for (int pixelLayerIndex = 0; pixelLayerIndex < count; pixelLayerIndex++)
{
if (pixelDatasReal[compareIndex][1].index == pixelDatas[index][pixelLayerIndex].index)
{
all_diff = false;
break;
}
}
if (all_diff)
{
SplatPixelData temp = pixelDatasReal[compareIndex][1];
pixelDatas[index].Add(temp);
count++;
}
}
//8 layer3
for (int k = 0; k < 8; k++)
{
int compareIndex = Mathf.Clamp(index + Mathf.FloorToInt(offsetTile[k].x + offsetTile[k].y * width), 0, 512 * 512 - 1);
if (pixelDatasReal[compareIndex].Count < 3)
continue;
bool all_diff = true;
for (int pixelLayerIndex = 0; pixelLayerIndex < count; pixelLayerIndex++)
{
if (pixelDatasReal[compareIndex][2].index == pixelDatas[index][pixelLayerIndex].index)
{
all_diff = false;
break;
}
}
if (all_diff)
{
SplatPixelData temp = pixelDatasReal[compareIndex][2];
pixelDatas[index].Add(temp);
count++;
}
}
colsRes3L[index] = EncodedIntoThreeLayer(pixelDatas[index]);
}
}

修改Blend图生成方式:
//save and sort the splat weight of every single pixel
for (int i = 0; i < width; i++)
{
for (int j = 0; j < width; j++)
{
List blendWeight = new List();
for (int index = 0; index < 16; index++)
{
if (index < prototypes.Length && splatData[i, j, index] > 0)
{
blendWeight.Add(new SplatPixelData(index, splatData[i, j, index], splatData[i, j, index]));
}
else
{
blendWeight.Add(new SplatPixelData(index, 0, 0));
}
}
blendWeight.Sort(ComparePixelData);
float iTotal = blendWeight[0].fxWeight + blendWeight[1].fxWeight + blendWeight[2].fxWeight;
//only save top three splat
for (int delete = 3; delete < 16; delete++)
{
blendWeight[delete] = new SplatPixelData(blendWeight[delete].index, 0, 0);
}
for (int s = 0; s < 4; s++)
{
Color resCol = new Color
(
(blendWeight.Find((SplatPixelData data) => data.index == 4 * s).fxWeight) / iTotal,
(blendWeight.Find((SplatPixelData data) => data.index == 4 * s + 1).fxWeight) / iTotal,
(blendWeight.Find((SplatPixelData data) => data.index == 4 * s + 2).fxWeight) / iTotal,
(blendWeight.Find((SplatPixelData data) => data.index == 4 * s + 3).fxWeight) / iTotal
);
int resIndex = (Mathf.FloorToInt(s / 2) * width + j) + (i + (int)(s % 2) * width) * doubleWidth;
blendCol[resIndex] = resCol;
}
}
}
